ll calculate_score() {
  ll last[26] = {};
  ll score = 0;
  ll sumc = 0;
  rep(i, 0, 26) sumc += c[i];
  ll mem_diff = 0;
  rep(d, 0, D) {
    score += s[d][t[d] - 1];
    mem_diff += -sumc + c[t[d] - 1] * (d + 1 - last[t[d] - 1]);
    score += mem_diff;
    last[t[d] - 1] = d + 1;
  }
  return score;
}
